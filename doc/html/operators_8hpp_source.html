50 inline double laplacian5_per_grid(double** u, int i, int j, int M, int N, float hsq) {
51 
52  double lap = -4*u[i][j];
53 
54  //u[i-1][]
55  if (i == 0)
56  lap += u[M-1][j];
57  else
58  lap += u[i-1][j];
59 
60  //u[i+1][]
61  if (i == M-1)
62  lap += u[0][j];
63  else
64  lap += u[i+1][j];
65 
66  // u[][j-1]
67  if (j == 0)
68  lap += u[i][N-1];
69  else
70  lap += u[i][j-1];
71 
72  // u[][j+1]
73  if (j == N-1)
74  lap += u[i][0];
75  else
76  lap += u[i][j+1];
77 
78  return lap / hsq;
79 }

339 std::vector<int> get_mutation_coords(char mscheme, double** v, int xlo, int xhi, int ylo, int yhi)
340 {
341  std::vector<int> coords(2,0);
342 
343  switch(mscheme) {
344 
345  // mutant at a maximum of the catalyst
346  case 'M': {
347  // look for a peak
348  double cmax = LOWLIM;
349  for (int x=xlo; x<=xhi; ++x)
350  for (int y=ylo; y<=yhi; ++y)
351  if (v[x][y] > cmax) {
352  cmax = v[x][y];
353  coords[0] = x;
354  coords[1] = y;
355  }
356  }
357  break;
358 
359  // mutant at a minimum of the catalyst
360  case 'm': {
361  // look for a peak
362  double cmin = 1.0;
363  for (int x=xlo; x<xhi; ++x)
364  for (int y=ylo; y<yhi; ++y)
365  if ((v[x][y] < cmin) && (v[x][y] > LOWLIM)) {
366  cmin = v[x][y];
367  coords[0] = x;
368  coords[1] = y;
369  }
370  }
371  break;
372 
373  // mutant at random position
374  case 'r': {
375  // set up random number generator
376  const gsl_rng_type * T;
377  T = gsl_rng_taus;
378  gsl_rng * rgen = gsl_rng_alloc(T);
379  if (!rgen)
380  fprintf(stderr, "Error allocating random number generator.\n");
381  gsl_rng_set(rgen, (unsigned)time(0));
382 
383  coords[0] = xlo + gsl_rng_uniform_int(rgen, xhi);
384  coords[1] = gsl_rng_uniform_int(rgen, yhi);
385  }
386  break;
387  }
388 
389  return coords;
390 }

405 void mutation_event(double** wt, double** mut, int M, int N, float mfract, int msize, char mscheme, bool m_internal)
406 {
407  float mconc = 0.0; // mutant concentration to place
408  int i,j; // position of mutant event
409 
410  // Get event coordinates by different schemes
411  std::vector<int> coords = get_mutation_coords(mscheme, wt, 0, M-1, 0, N-1);
412  i = coords[0];
413  j = coords[1];
414 
415  // Place mutant drop
416  if (m_internal)
417  mconc = mfract * wt[i][j];
418  else
419  mconc = mfract;
420 
421  mut[i][j] = mconc;
422  if (m_internal) {
423  // converted catalyst
424  wt[i][j] -= mconc;
425  if (wt[i][j] < LOWLIM)
426  wt[i][j] = 0.0;
427  }
428 
429  // periodic boundary conditions
430  if (msize)
431  for (int x= -msize; x<= msize; ++x)
432  for (int y= -msize; y<= msize; ++y) {
433 
434  mut[(i+M+x)%M][(j+N+y)%N] = mconc;
435 
436  if (m_internal) {
437  // convert catalyst
438  wt[(i+M+x)%M][(j+N+y)%N] -= mconc;
439  if (wt[(i+M+x)%M][(j+N+y)%N] < LOWLIM)
440  wt[(i+M+x)%M][(j+N+y)%N] = 0.0;
441  }
442  }
